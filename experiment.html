class Rating:
    def __init__(self, objects, rating):
        self.objects = objects
        self.rate = {obj:rating[ind] for ind, obj in enumerate(objects)}
        self.rating = rating
    
    def printRating(self):
        ratingSort = self.getRatingSort()        
        for obj, r in ratingSort:
            print obj
            
    def getRatingSort(self):
        ratingSort = [(self.objects[ind], r) for ind, r in enumerate(self.rating)]
        ratingSort.sort(key=lambda x: x[1], reverse=True)
        return ratingSort

def metrikaKemeni(rating1, rating2):
    comp = lambda a, b: bool(a > b) - bool(a < b)
    result = 0.0
    for obj1 in rating1.objects:
        for obj2 in rating2.objects:            
            result += abs(comp(rating1.rate[obj1], rating1.rate[obj2]) - comp(rating2.rate[obj1], rating2.rate[obj2]))
    return result / 2

def diffKemeni(S, ratings):
    result = 0.0
    for rating in ratings:
        result += metrikaKemeni(rating, S)    
    return result

def medianaKemeni(ratings):
    objects = ratings[0].objects
    minResult = sys.maxint
    bestKemeni = None
    for perm in itertools.permutations(range(len(objects))):        
        kemeni = Rating(objects, perm)
        result = diffKemeni(kemeni, ratings)    
        if result < minResult:
            bestKemeni = kemeni
            minResult = result
            bestKemeni.diff = result
    return bestKemeni

def POF(matrix):
    numObjects = len(matrix[0])
    obj = set(range(numObjects))
    waves = []
    def dom(a, b):
        return allOld(row[a] <= row[b] for row in matrix) and anyOld(row[a] < row[b] for row in matrix)
        
    while obj: 
        waves.append(set())
        domNotAnyWas = True
        for i in range(numObjects):
            domNotWas = True
            if i not in obj:
                continue
            for j in range(numObjects):
                if i == j or j not in obj:
                    continue                
                if dom(i, j):
                    domNotWas = False
                    break
            if domNotWas:
                domNotAnyWas = False
                waves[-1].add(i)                            
        if domNotAnyWas:
            waves[-1] = obj
            break
        for i in waves[-1]:
            obj.remove(i)
    return waves

def myAlgo(matrix, objects, log=False):
    waves = POF(matrix)      
    superRating = {}
    size = 0
    for wave in waves:
        waveList = list(wave)                
        localKemeni = medianaKemeni([Rating(waveList, [row[obj] for obj in wave]) for row in matrix])
        if log:
            print len(waveList)        
        superRating.update({obj_rate[0]:ind+size+1 for ind, obj_rate in enumerate(localKemeni.getRatingSort())})
        size += len(waveList)
    
    ratings = [Rating(objects, rate) for rate in matrix]
    myRes = Rating(objects, [-superRating[ind] for ind, obj in enumerate(objects)])
    myRes.diff = diffKemeni(myRes, ratings)
    
    return myRes

def checkerTime(f):
    start_time = time.time()
    result = f()
    return result, time.time() - start_time

def getKorelation(ratings):
    res = 0
    for i in ratings:
        for j in ratings:
            if i == j:
                continue
            res += metrikaKemeni(i, j)
    n = len(ratings)
    return res / (n * (n-1))
