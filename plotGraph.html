def myPlot(*args, **kwargs):
    name = None
    if 'name' in kwargs:
        name = kwargs['name']
        del kwargs['name']
    plt.plot(*args, **kwargs)
    if name:
        plt.savefig('images/{0}.eps'.format(name), format='eps', dpi=1000)
    plt.show()
